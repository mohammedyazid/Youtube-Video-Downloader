self.SUBMIT.clicked.connect(self.submitc)
    
    
    
    def submitc(self):
        self.link = self.URL.toPlainText()
        self.yt = YouTube(self.link)
        views = str(self.yt.views)
        length = str(self.yt.length)
        rating = str(self.yt.rating)
        #Showing details
        self.title.setText(self.yt.title)
        self.views.setText(views)
        self.length.setText(length)
        self.rating.setText(rating)
        #Getting the highest resolution possible
        self.ys = self.yt.streams.get_highest_resolution()
        
        #Starting download
        self.status.setText("Downloading...")
        self.ys.download()
        self.status.setText("Download completed!!")
